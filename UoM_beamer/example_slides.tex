\begin{frame}{Model performance}
  \begin{itemize}
    \item Model converges rapidly
    \item Catalogue-weighted model's \textbf{sensitivity} outperformed or matched TB-Profiler (baseline) - $13/15$ drugs
    \item For 14/15 drugs, \textbf{sensitivity} \textit{improved} after training and threshold application
    \item There was a trade-off in \textbf{specificity}, with a floor of 0.8 during threshold sweeps
      \begin{itemize}
        \item $\uparrow$ Sens. $\implies$ $\uparrow$ TPR $\implies$ $\downarrow$ Missed R cases
        \item $\downarrow$ Spec. $\implies$ $\uparrow$ FPR $\implies$ $\uparrow$ Missed S cases
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Ablation biological significance}
  \begin{itemize}
    \item In $3/4$ 1st-line drugs, SNPs found that \textit{improved} \textbf{sensitivity} (compared to non-ablated baseline)
      \begin{itemize}
        \item \textbf{H} - \textit{katG} and \textit{rpoB} 
        \item \textbf{R} - \textit{rpoB} (drug target gene)
        \item \textbf{E} - \textit{rpoB} , \textit{katG} , \textit{embB} (drug target gene)
      \end{itemize}
    \item For 2nd-line drugs:
      \begin{itemize}
        \item \textbf{FQs} - Both LFX and MFX found ablated SNPs in \textit{gyrA} (drug target gene)
        \item \textbf{CFZ} - Ablated SNPs of unknown function 
        \item \textbf{ETO} - \textit{inhA} T>G, possible \textbf{cross-resistance} 
      \end{itemize}
    \item Potential confounders - \textit{rpoB} C>T, \textit{katG} C>G 
      \end{itemize}
\end{frame}

\begin{frame}{Candidate Mutations}
  \begin{itemize}
    \item 1st-line drugs $\rightarrow$ highly populated plots (top-right quad.)
      \begin{itemize}
        \item Selection pressure over decades of use
        \item Possible compensatory mutations 
      \end{itemize}
    \item 2nd-line drugs $\rightarrow$ sparsely populated (with exceptions)
      \begin{itemize}
        \item Class imbalance $\rightarrow$ Cannot pass filters
        \item Newer drugs (BDQ, LZD) $\rightarrow$ Few R samples
        \item \textbf{Exception} - ETO $\rightarrow$ INH-ETO cross-resistance, linked mechanisms
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Limitations}
  \begin{itemize}
    \item Training data lacks resistant cases for newer drugs
      \begin{itemize}
        \item Class imbalances for BDQ, LZD, DLM, CFZ
        \item No data for TD, PA, or PTO
      \end{itemize}
    \item Model rapidly converges $\rightarrow$ rapidly hits \textbf{representational capacity}
      \begin{itemize}
        \item May not learn cross-resistance or pleiotropic R mechanisms
      \end{itemize}
    \item SNP filtering needs to be more robust
    \begin{itemize}
      \item Flag and remove confounders
      \item Comprehensive screening and flagging of lineage patterns 
    \end{itemize}
  \item Possible relaxation of filtering significance threshold
    \begin{itemize}
      \item $-\log_{10}(p) \geq 5$ - maybe too stringent for newer drugs
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Next steps}
  \begin{itemize}
    \item \textit{Update} the dataset for newer drugs
    \begin{itemize}
      \item Phenotypic R data for TD, PA, and PTO
      \item More R data for BDQ, LZD, DLM, and CFZ
      \begin{itemize}
        \item BPaLM - promising 
      \end{itemize}
    \end{itemize}
    \item Validation from external cohort (CRyPTIC consortium, MIRU-VNTR)
    \item Lineage stratification for evaluation 
    \item Wet-lab validation of candidate SNPs $\rightarrow$ possibly included in future \textbf{catalogues} or \textbf{AMR prediction tools}
    \item Implement different network architecture (Hidden layers for shared mechanisms (ETO-INH), etc.)
  \end{itemize}
\end{frame}
